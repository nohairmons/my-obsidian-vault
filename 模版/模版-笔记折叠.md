<%*
// 1. 获取当前编辑器对象
const view = app.workspace.getActiveViewOfType(tp.obsidian.MarkdownView);
if (!view) return;
const editor = view.editor;

// 2. 获取你用鼠标选中的文本
const selection = editor.getSelection();
if (!selection) return;

// 3. 将选中的文本按行切分，并在每行开头加上 "> "
const calloutLines = selection.split('\n').map(line => '> ' + line).join('\n');

// 4. 组装成完整的折叠 Callout 格式
const finalResult = `> [!note]- 点击展开折叠内容\n${calloutLines}`;

// 5. 直接用底层 API 替换选中区域，完美避开 Templater 自身的语法解析报错
editor.replaceSelection(finalResult);
-%>
